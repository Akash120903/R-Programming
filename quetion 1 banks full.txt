--Q1 DBMS
CREATE TABLE customer (
    cust_id INT PRIMARY KEY,
    name VARCHAR(50),
    income_permonth DECIMAL(10, 2),
    gender VARCHAR(10),
    location_pincode INT
    
);

CREATE TABLE accounts (
    account_number INT PRIMARY KEY,
    cust_id INT,
    account_type VARCHAR(20),
    balance_inlakhs DECIMAL(10, 2),
    ifsc_code VARCHAR(10)
);


CREATE TABLE branch (
    ifsc_code VARCHAR(10) PRIMARY KEY,
    branch_name VARCHAR(50),
    location_pincode INT
);


CREATE TABLE city (
    location_pincode INT PRIMARY KEY,
    cityname VARCHAR(50),
    state VARCHAR(50)
);

--1
SELECT COUNT(*) AS num_accounts
FROM accounts
WHERE balance_inlakhs > 1 AND cust_id IN (
    SELECT cust_id
    FROM customer
    WHERE location_pincode = (
        SELECT location_pincode
        FROM city
        WHERE cityname = 'City A'
    )
);

--2
SELECT COUNT(*) AS num_branches
FROM branch
WHERE location_pincode = (
    SELECT location_pincode
    FROM city
    WHERE cityname = 'City B'
) AND ifsc_code IN (
    SELECT ifsc_code
    FROM accounts
    WHERE cust_id IN (
        SELECT cust_id
        FROM customer
        WHERE gender = 'Female'
    )
    GROUP BY ifsc_code
    HAVING COUNT(*) > 2
);

--3
SELECT name
FROM customer
WHERE cust_id IN (
    SELECT cust_id
    FROM accounts
    WHERE ifsc_code = (
        SELECT ifsc_code
        FROM branch
        WHERE branch_name = 'Branch B'
    )
    GROUP BY cust_id
    HAVING COUNT(*) > 1
);

--4
SELECT cityname
FROM city
WHERE location_pincode IN (
    SELECT location_pincode
    FROM accounts
    GROUP BY location_pincode
    HAVING AVG(balance_inlakhs) = (
        SELECT MIN(avg_balance)
        FROM (
            SELECT AVG(balance_inlakhs) AS avg_balance
            FROM accounts
            GROUP BY location_pincode
        )
    )
);
